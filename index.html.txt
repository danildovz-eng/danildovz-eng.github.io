    function drawChart(pts, avgPrice, ownsStock) { 
        const svg = document.getElementById('stock-chart-svg'); const min = Math.min(...pts) * 0.99; const max = Math.max(...pts) * 1.01; const width = 100; const height = 100; const stepX = width / (pts.length - 1); 
        let strokeColor = '#238636'; if (pts[pts.length-1] < pts[pts.length-2]) strokeColor = '#DA3633'; else if (pts[pts.length-1] === pts[pts.length-2]) strokeColor = '#D29922'; 
        let d = `M 0 ${height - ((pts[0] - min)/(max - min) * height)}`; for(let i=1; i<pts.length; i++) { d += ` L ${i * stepX} ${height - ((pts[i] - min)/(max - min) * height)}`; } 
        let html = `<path d="${d}" fill="none" stroke="${strokeColor}" stroke-width="2" vector-effect="non-scaling-stroke"/>`; 
        if(ownsStock && avgPrice > 0) { let yPos = height - ((avgPrice - min)/(max - min) * height); if(yPos >= 0 && yPos <= 100) { html += `<line x1="0" y1="${yPos}" x2="100" y2="${yPos}" stroke="#8B949E" stroke-width="1" stroke-dasharray="2,2" vector-effect="non-scaling-stroke"/><text x="2" y="${yPos - 2}" font-size="4" fill="#8B949E" style="font-family:sans-serif;">Куплено: ${avgPrice.toFixed(2)}</text>`; } } 
        svg.innerHTML = html; 
    }

    window.makeRpsMove = function(move) {
        if(!rpsCurrentRoomId) return;
        document.getElementById('rps-game-status').innerText = "Ожидание хода соперника...";
        document.getElementById('rps-game-status').style.color = "var(--text-muted)";
        
        database.ref('rps_rooms/' + rpsCurrentRoomId).once('value').then(snap => {
            let r = snap.val();
            if(!r) return;
            if(r.p1 === currentUserKey) database.ref('rps_rooms/' + rpsCurrentRoomId).update({ p1Move: move });

        });

    function listenToCurrentGame() {
        database.ref('rps_rooms/' + rpsCurrentRoomId).on('value', snap => {
            let r = snap.val();
            if(!r) { 
                if(document.getElementById('rps-modal').classList.contains('active')) {
                    document.getElementById('rps-lobby-view').style.display = 'block';
                    document.getElementById('rps-game-view').style.display = 'none';
                    rpsCurrentRoomId = null;
                }
                return; 
            }
            if(r.p1Move && r.p2Move && r.status !== 'finished') {
                let winner = 0;
                if(r.p1Move === r.p2Move) winner = 0;
                else if((r.p1Move==='rock'&&r.p2Move==='scissors') || (r.p1Move==='paper'&&r.p2Move==='rock') || (r.p1Move==='scissors'&&r.p2Move==='paper')) winner = 1;
                else winner = 2;
                
                database.ref('rps_rooms/' + rpsCurrentRoomId).update({ status: 'finished', winner: winner });
                progressQuest('casino');
                
                const user = getState();
                let amI_P1 = r.p1 === currentUserKey;
                let myWin = (amI_P1 && winner === 1) || (!amI_P1 && winner === 2);
                let tie = winner === 0;

                setTimeout(() => {
                    if(tie) {
                        user.balance += r.bet; addTransaction("КНБ (Ничья)", 0, '✌️'); showToast("Ничья", "info");
                    } else if(myWin) {
                        let winAmt = r.bet * 2; user.balance += winAmt; addTransaction("КНБ (Победа)", winAmt - r.bet, '✌️'); showToast("Вы победили!", "success"); showSuccessEffect();
                    } else {
                        addTransaction("КНБ (Поражение)", -r.bet, '✌️'); showToast("Вы проиграли", "error");
                    }
                    saveDB(); updateBalanceDisplayQuiet();
                    database.ref('rps_rooms/' + rpsCurrentRoomId).remove();
                    closeRpsModal();
                }, 1000);
            }
        });
    }

    function checkQuests() { 
        const user = getState(); const today = new Date().toLocaleDateString('ru-RU'); 
        if(!user.quests || user.quests.date !== today || !user.quests.list) { 
            let shuffled = [...questTemplates].sort(() => 0.5 - Math.random()).slice(0, 3);
            let activeList = shuffled.map(q => ({id: q.id, icon: q.icon, desc: q.desc, target: q.target, progress: 0}));
            user.quests = { date: today, list: activeList, done: false }; 
        } 
    }
    
    function renderQuests() {
        const user = getState(); if(!user) return; checkQuests(); const q = user.quests; const container = document.getElementById('quest-container'); if(!container) return;
        if(q.done) { container.innerHTML = '<div style="color:var(--success); font-weight:600; text-align:center;">Задания выполнены</div>'; return; }
        let html = ''; let allDone = true;
        q.list.forEach(item => {
            let prog = Math.min(item.progress, item.target);
            if(prog < item.target) allDone = false;
            html += `<div style="font-size:13px; margin-bottom:8px; display:flex; justify-content:space-between;"><span>${item.icon} ${item.desc}</span> <span style="color:var(--text-muted);">${prog}/${item.target}</span></div>`;
        });
        if(allDone) { html += `<button class="btn-main btn-success" style="margin-top:10px;" onclick="claimQuests()">Получить награду (500 $)</button>`; }
        container.innerHTML = html;
    }

                let receiverKey = Object.keys(snapshot.val())[0]; let receiverData = snapshot.val()[receiverKey];
                database.ref('users/' + receiverKey + '/incoming_transfers').push({ amount: amount, senderName: sender.name, date: new Date().toLocaleString('ru-RU', {day: '2-digit', month: 'short', hour: '2-digit', minute: '2-digit'}) }).then(() => { 
                    sender.balance -= amount; sender.history.unshift({ icon: '💸', title: `Перевод ${receiverData.name}`, date: new Date().toLocaleString('ru-RU', {day: '2-digit', month: 'short', hour: '2-digit', minute: '2-digit'}), amount: -amount, type: 'minus' }); saveDB(); 
                    document.getElementById('transfer-card').value = ''; document.getElementById('transfer-amount').value = ''; btn.innerText = "Отправить средства"; btn.disabled = false; updateBalanceDisplayQuiet(); renderHistory(); switchTab('screen-dashboard', document.getElementById('nav-item-home')); showPush("Перевод доставлен", `Адресат: <b>${receiverData.name}</b>`, "💸"); 
                }).catch(() => { showToast("Ошибка сети", "error"); btn.innerText = "Отправить средства"; btn.disabled = false; });
            } else { 
                sender.balance -= amount; sender.history.unshift({ icon: '💸', title: `Перевод *${cardInput.slice(-4)}`, date: new Date().toLocaleString('ru-RU', {day: '2-digit', month: 'short', hour: '2-digit', minute: '2-digit'}), amount: -amount, type: 'minus' }); saveDB(); 
                document.getElementById('transfer-card').value = ''; document.getElementById('transfer-amount').value = ''; btn.innerText = "Отправить средства"; btn.disabled = false; updateBalanceDisplayQuiet(); renderHistory(); switchTab('screen-dashboard', document.getElementById('nav-item-home')); showPush("Перевод выполнен", `Сумма: <b>${amount} $</b>`, "💸"); 
            }
        }).catch(err => { showToast("Ошибка связи", "error"); btn.innerText = "Отправить средства"; btn.disabled = false; });
    }

    window.addEventListener('load', checkAuthState);
